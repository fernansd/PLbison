# Prueba estructuras de control
dato := 10;
escribir(dato);
mientras (dato > 0)
hacer {
  si (dato _mod 2 == 0) entonces escribir(dato)
  si_no escribir(-dato)
  fin_si;
  dato := dato - 1;
} fin_mientras;

si ((dato := 3) ==  4)
entonces {
	escribir(350);
} si_no {
	escribir(750);
} fin_si;

dato := 5;
mientras (dato > 0)
hacer {
	escribir(dato);
	dato := dato - 1;
} fin_mientras;

dato := 0;
repetir {
	escribir(dato);
	dato := dato + 1;
} hasta (dato == 5);

dato := 8;
escribir(dato);
para dato
	desde 0
	hasta 6
	paso 2
	hacer {
		escribir(dato);
} fin_para;
